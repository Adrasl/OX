32 // Function: define_key_t
33 //
34 // *Brief*
35 // Templatized version of PandaFramework::define_key. Will create at compilation
36 // time the static stub function that calls the member function func of object
37 // obj.
38 //
39 // *Parameters*
40 // T class of obj, see that parameter below.
41 // func member function to callback, i.e. T::func, upon the key event.
42 // The signature of the member function must be:
43 // void T::foo(const Event* event)
44 // wf the Panda3d window posting the events.
45 // event_name same parameter as in PandaFramework::define_key.
46 // description same parameter as in PandaFramework::define_key.
47 // obj the object affected by the key event.
48 //
49 // *Example*
50 // For class World with member function move_forward that is to be called each
51 // time key `w' is pressed:
52 //
53 // World::World(WindowFramework* wf)
54 // {
55 // ...
56 // define_key_t<World, &World::move_forward>(wf, "w", "MoveForward", this);
57 // ...
58 // }
59 //
60 // void World::move_forward(const Event* event)
61 // {
62 // ...
63 // }
64 //
65 // Can be simplified further using a macro to specify constant parameters
66 // (assuming World has a data member m_wf that points to the relevant
67 // WindowFramework*):
68 //
69 // #define WORLD_DEFINE_KEY(event_name, description, func)
70 // define_key_t<World, &World::func>(m_wf, event_name, description, this)
71 //
72 // Then the call above would be reduced to:
73 //
74 // WORLD_DEFINE_KEY("w", "MoveForward", move_forward);
75 //
76 template <class T, void (T::*func)(const Event* event)>
77 inline void define_key_t(WindowFramework* wf,
78  const string& event_name,
79  const string& description,
80  T* obj)
81  {
82  // preconditions
83  if(wf == NULL)
84  {
85  nout << "ERROR: parameter wf cannot be null." << endl;
86  return;
87  }
88 
89  // keyboard needs to be enabled and since it does not hurt to enable it
90  // multiple times, just enable it at each call.
91  wf->enable_keyboard();
92  // at the compilation, an EventCallbackStub for T::func will be generated
93  wf->get_panda_framework()->define_key(event_name,
94  description,
95  &EventCallbackStub<T, func>,
96  obj);
97  }

101 //
102 // *Brief*
103 // Templatized version of AsyncTaskManager::add. Will create at compilation
104 // time the static stub function that calls the member function func of object
105 // obj.
106 //
107 // *Parameters*
108 // T class of obj, see that parameter below.
109 // func member function the task manager has to callback, i.e. T::func.
110 // The signature of the member function must be:
111 // AsyncTask::DoneStatus T::foo(GenericAsyncTask *task)
112 // name same parameter as in GenericAsyncTask::GenericAsyncTask.
113 // obj the object affected by the key event.
114 //
115 // *Example*
116 // For class World with member function audio_loop that is to be called each
117 // frame:
118 //
119 // PT(GenericTask) task =
120 // add_task_t<World, &World::audio_loop>("AudioLoop", this);
121 //
122 // Can be simplified further using a macro to specify constant parameters:
123 //
124 // #define WORLD_ADD_TASK(name, func)
125 // add_task_t<World, &World::func>(name, this)
126 //
127 // Then the call above would be reduced to:
128 //
129 // PT(GenericTask) task = WORLD_ADD_TASK("AudioLoop", audio_loop);
130 //
131 template<class T, AsyncTask::DoneStatus (T::*func)(GenericAsyncTask *task)>
132 inline GenericAsyncTask* add_task_t(const string &name, T* obj)
133  {
134  // create the new task. A TaskStub for T::func will be generated.
135  PT(GenericAsyncTask) task = new GenericAsyncTask(name,
136  &TaskStub<T, func>,
137  obj);
138  // register the new task with the global AsyncTaskManager
139  AsyncTaskManager::get_global_ptr()->add(task);
140  // return the task so the user can control it
141  return task;
142  }

176 template <class T, void (T::*func)(const Event* event)>

178  {
179  return EventHandler::get_global_event_handler()->add_hook(
